\begin{frame}{einfacher Einstieg}
        \onslide
            Zu zeigen: Schnitt ist kommutativ, d.h. $A \cap B = B \cap A$
        \begin{columns}
        \column{0.5\textwidth}
        \onslide{
            \begin{align*}
                \text{\quotedblbase}\implies\text{\textquotedblright}:\\
                x\in A \cap B &\implies x\in A \wedge x \in B\\
                &\implies x\in B \wedge x \in A\\
                &\implies x\in B \cap A
            \end{align*}
            }
        \column{0.5\textwidth}
        \onslide{
            \begin{align*}
                \text{\quotedblbase}\impliedby\text{\textquotedblright}:\\
                x\in B \cap A &\implies x\in B \wedge x \in A\\
                &\implies x\in A \wedge x \in B\\
                &\implies x\in A \cap B
            \end{align*}
            }
        \end{columns}
        \qed\\
    \small{\emph{Anmerkung:} $\wedge$ ist kommutativ}
\end{frame}

\begin{frame}{einfacher Einstieg}
        \onslide
            Zu zeigen: $A \setminus (B \cup C) = (A \setminus B) \cap (A \setminus C)$
        \onslide{
            \begin{align*}
                x\in A \setminus (B \cup C) &\only<0-7|handout:0>{\implies}\only<8-|handout:1>{\text{\alert<8|handout:0>{$\iff$}}} x\in A \wedge \neg (x \in B \cup C)\\
                &\only<0-6>{\implies|handout:0}\only<7-|handout:1>{\text{\alert<7|handout:0>{$\iff$}}} x\in A \wedge \neg (x \in B \vee x \in C)\\
                &\only<0-5>{\implies|handout:0}\only<6-|handout:1>{\text{\alert<6|handout:0>{$\iff$}}} x \in A \wedge \neg (x \in B) \wedge \neg (x \in C)\\
                &\only<0-4>{\implies|handout:0}\only<5-|handout:1>{\text{\alert<5|handout:0>{$\iff$}}} x \in A \wedge \neg (x \in B) \wedge x \in A \wedge \neg (x \in C)\\
                &\only<0-3>{\implies|handout:0}\only<4-|handout:1>{\text{\alert<4|handout:0>{$\iff$}}} (x \in A \wedge \neg (x \in B)) \wedge (x \in A \wedge \neg (x \in C))\\
                &\only<0-2>{\implies|handout:0}\only<3-|handout:1>{\text{\alert<3|handout:0>{$\iff$}}} (x \in A \setminus B) \wedge (x \in A \setminus C)\\
                &\only<0-1>{\implies|handout:0}\only<2-|handout:1>{\text{\alert<2|handout:0>{$\iff$}}} x \in (A \setminus B) \cap (A \setminus C)
            \end{align*}\qed
            }
        \\
    \small{\emph{Rechenregel:} $\neg (A \wedge B) \iff \neg A \vee \neg B$, \\ \hspace{1.9cm}$\neg (A \vee B) \iff \neg A \wedge \neg B$}
\end{frame}

{\setbeamercolor{palette primary}{bg=ExColor}
\begin{frame}<handout:0>[fragile]{Lösungen}
\onslide Zu zeigen: $A=\overline{\overline{A}}$
    \onslide{
    \begin{align*}
        x\in\overline{\overline{A}}
        &\iff\neg(x\in\overline{A})\\
        &\iff\neg(\neg (x\in A))\\
        &\iff x\in A
    \end{align*}\qed
    }
\end{frame}
}

{\setbeamercolor{palette primary}{bg=ExColor}
\begin{frame}<handout:0>[fragile]{Lösungen}
    \onslide Zu zeigen: $A \cap B=\overline{(\overline{A}\cup\overline{B})}$
    \onslide{
    \begin{align*}
        x \in \overline{(\overline{A} \cup \overline{B})}
        &\iff \neg(x \in \overline{A} \cup \overline{B})
        \\&\iff \neg(x \in \overline{A} \vee x \in \overline{B})
        \\&\iff \neg(\neg(x \in A) \vee \neg(x \in B))
        \\&\iff \neg(\neg(x\in A))\wedge\neg(\neg(x\in B))
        \\&\iff x \in A \wedge x \in B
        \\&\iff x \in A \cap B
    \end{align*}\qed
    }
\end{frame}
