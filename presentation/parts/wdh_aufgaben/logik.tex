% Aufgaben zu Aussagenlogik und Quantoren

{\setbeamercolor{palette primary}{bg=ExColor}
\begin{frame}[fragile]{Aufgaben zu Logik}
    \begin{alertblock}{Aufgaben}
        Bestimme den Wahrheitswert der folgenden Aussagen
    \end{alertblock}
    \metroset{block=fill}
    \begin{block}{Normal}
        \begin{itemize}
            \item $\neg (42 = 11) \land (0\in \mathbb{N} \lor 0 = |\mathbb{N}|)$
            \item $\forall x \in \mathbb{Q}\ \exists p,q \in \mathbb{Z}: \left(\left(x = \frac{p}{q} \land q = 1\right) \implies \frac{p}{q}\in \mathbb{Z}\right)$
            \item $\exists n \in \mathbb{N}: (n \equiv 13 \pmod 2025 \implies n < 2025) \iff 14 \subset \mathbb{N}$
            \item $(\exists n \in (\mathbb{N} \cap \emptyset): |n| \geq 0) \iff (A\cap B = B\cap A)$
        \end{itemize}
    \end{block}
    \metroset{block=fill}
    \begin{block}{Schwer}
        \begin{itemize}
            \item $\forall x \in \mathbb{Z}: x \equiv 0 \pmod{2} \vee x \equiv 1 \pmod{4}$
        \end{itemize}
    \end{block}
\end{frame}
}

{\setbeamercolor{palette primary}{bg=ExColor}
\begin{frame}<handout:0>[fragile]{Lösungen}
    \only<1-4>{

        \begin{block}{Normal}
            \begin{align*}
                \only<1>{\neg (42 = 11) \land (0\in \mathbb{N} \lor 0 = |\mathbb{N}|)}

                \only<3>{\exists n \in \mathbb{N}: (n \equiv 13 \mod 2025 \implies n < 2025) \iff 14 \subset \mathbb{N}}
                \only<4>{(\exists n \in (\mathbb{N} \cap \emptyset): |n| \geq 0) \iff (A\cap B = B\cap A)}
            \end{align*}
        \end{block}
        \begin{itemize}[<+- | alert@+>]
            \item wahr
            \item wahr
            \item nicht zulässig
            \item falsch
        \end{itemize}
    }
    \only<5>{
        \metroset{block=fill}
        \begin{block}{Schwer}
            \begin{align*}
                \only<5>{\forall x \in \mathbb{Z}: x \equiv 0 \pmod{2} \vee x \equiv 1 \pmod{4}}
            \end{align*}

    }
