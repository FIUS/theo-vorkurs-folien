                  $V_2=\{S, A\}$,
                  $P_2 = \{S \rightarrow aA | c | d,\ A \rightarrow bS\}$\\}
        \item<3->Automat: \alert<3>{$M_2 = \left(\left\{q_0, q_1, q_2\only<4->{, \alert<4>{q_f}}\right\}, \Sigma_2, \delta_2, q_0, \left\{q_2\right\}\right)$}\\

              \end{columns}
